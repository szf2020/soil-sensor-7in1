82// –°–æ–≤—Ä–µ–º–µ–Ω–Ω–∞—è C++ –∞–ª—å—Ç–µ—Ä–Ω–∞—Ç–∏–≤–∞ —Å template parameter pack
83template <typename... Args>
84String formatLogMessageSafe(const char* format, Args&&... args)
85{
86 // Define once to avoid duplication
87 auto toStringAny = [](const auto& value) -> String {
88 using T = typename std::decay<decltype(value)>::type;
89 if constexpr (std::is_same<T, String>::value)
90 {
91 return value;
92 }
93 else if constexpr (std::is_same<T, const char*>::value || std::is_same<T, char*>::value)
94 {
95 return String(value ? value : "");
96 }
97 else if constexpr (std::is_same<T, bool>::value)
98 {
99 return value ? String("true") : String("false");
100 }
101 else if constexpr (std::is_arithmetic<T>::value)
102 {
103 return String(value);
104 }
105 else if constexpr (std::is_pointer<T>::value)
106 {
107 // Render pointer address as hex
108 uintptr_t addr = reinterpret_cast<uintptr_t>(value);
109 char buf[3 + (sizeof(uintptr_t) * 2)];
110 // 0x + hex digits, ensure null-termination
111 int n = snprintf(buf, sizeof(buf), "0x%llx", static_cast<unsigned long long>(addr));
112 (void)n;
113 return String(buf);
114 }
115 else
116 {
117 return String(value);
118 }
119 };
120
121 // If format is null, just join arguments
122 if (format == nullptr)
123 {
124 String out;
125
126 auto appendArg = [&](const auto& v) {
127 if (out.length() > 0) out += " ";
128 out += toStringAny(v);
129 };
130 (appendArg(std::forward<Args>(args)), ...);
131 return out;
132 }
133
134 // Count real printf-style placeholders, ignoring escaped percent signs ("%%").
135 auto countPrintfSpecifiers = [](const char* fmt) -> int {
136 int count = 0;
137 for (const char* p = fmt; *p; ++p)
138 {
139 if (*p != '%')
140 {
141 continue;
142 }
143
144 // Handle escaped percent: "%%" ‚Üí skip both
145 if (*(p + 1) == '%')
146 {
147 ++p; // skip the second '%'
148 continue;
149 }
150
151 // We are at start of a conversion specifier. Skip flags/width/precision/length.
152 ++p;
153 // Flags: -+ #0 '
154 while (*p == '-' || *p == '+' || *p == ' ' || *p == '#' || *p == '0' || *p == '\'')
155 {
156 ++p;
157 }
158 // Width: number or '*'
159 if (*p == '*')
160 {
161 ++p;
162 }
163 else
164 {
165 while (*p >= '0' && *p <= '9') ++p;
166 }
167 // Precision: .number or .*
168 if (*p == '.')
169 {
170 ++p;
171 if (*p == '*')
172 {
173 ++p;
174 }
175 else
176 {
177 while (*p >= '0' && *p <= '9') ++p;
178 }
179 }
180 // Length modifiers (subset sufficient for our targets)
181 if ((*p == 'h' && *(p + 1) == 'h') || (*p == 'l' && *(p + 1) == 'l'))
182 {
183 p += 2;
184 }
185 else if (*p == 'h' || *p == 'l' || *p == 'j' || *p == 'z' || *p == 't' || *p == 'L')
186 {
187 ++p;
188 }
189 // Conversion specifier
190 if (*p == '\0')
191 {
192 break;
193 }
194 const char c = *p;
195 switch (c)
196 {
197 case 'd': case 'i': case 'u': case 'o': case 'x': case 'X':
198 case 'f': case 'F': case 'e': case 'E': case 'g': case 'G': case 'a': case 'A':
199 case 'c': case 's': case 'p': case 'n':
200 ++count;
201 break;
202 default:
203 // Unknown specifier; treat as non-specifier (do not increment)
204 break;
205 }
206 }
207 return count;
208 };
209
210 const int placeholderCount = countPrintfSpecifiers(format);
211
212 // If there are no placeholders, or argument count mismatches, prefer safe concatenation
213 if (placeholderCount == 0 || placeholderCount != static_cast<int>(sizeof...(args)))
214 {
215 // Special sentinel: "\1" often used to mean "no base text, just args"
216 // In C/C++ the literal "\1" is a single char with value 1
217 const bool isSentinel = (format[0] == '\x01' && format[1] == '\0');
218 String out = isSentinel ? String("") : String(format);
219
220 if constexpr (sizeof...(args) > 0)
221 {
222 if (out.length() > 0) out += " ";
223 auto appendArg = [&](const auto& v) {
224 if (out.length() > 0 && out[out.length() - 1] != ' ') out += " ";
225 out += toStringAny(v);
226 };
227 (appendArg(std::forward<Args>(args)), ...);
228 }
229 return out;
230 }
231
232 std::array<char, 512> buffer{};
233 int result = snprintf(buffer.data(), buffer.size(), format, std::forward<Args>(args)...);
234 if (result < 0)
235 {
236 return String("LOG FORMAT ERROR");
237 }
238 if (result < static_cast<int>(buffer.size()))
239 {
240 return String(buffer.data());
241 }
242 buffer[buffer.size() - 1] = '\0';
243 return String(buffer.data());
244}
