82{

84 int result = snprintf(buffer.data(), buffer.size(), format, std::forward<Args>(args)...);
85 if (result < 0)

87 return String("LOG FORMAT ERROR");
88 }
89 if (result < static_cast<int>(buffer.size()))
90 {
91 return String(buffer.data());
92 }

95}
